num1 = float(input("Enter first number:")) 	//Default input string convert to float
op = input("Enter Operator: ")
num2 =  float(input("Enter second number")) 	//Default input string convert to float

if op       ==  "+":
	print(num1 + num2)

elif  op   ==  "-":
	print(num1 - num2)
elif  op   ==  "/"
	print(num1 / num2 )
elif  op   =="*"
	print(num1*num2)
